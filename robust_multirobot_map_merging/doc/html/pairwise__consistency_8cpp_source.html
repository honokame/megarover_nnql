7 Eigen::MatrixXi PairwiseConsistency::computeConsistentMeasurementsMatrix() {
8  // Determination of the chi squared threshold (numbers from chi-squared table)
9  double threshold;
10  if (nb_degree_freedom_ == 3){
11  threshold = 0.58;
12  } else {
13  threshold = 2.20;
14  }
15 
16  // Preallocate consistency matrix
17  Eigen::MatrixXi consistency_matrix(loop_closures_.size(), loop_closures_.size());
18 
19  // Iterate on loop closures
20  size_t u = 0;
21  for (const auto& loop_closure_1: loop_closures_) {
22  size_t v = 0;
23  for (const auto& loop_closure_2: loop_closures_) {
24  if (u < v) {
25  // Extract pose indexes
26  size_t i,j,k,l;
27  i = loop_closure_1.first;
28  k = loop_closure_1.second;
29  j = loop_closure_2.first;
30  l = loop_closure_2.second;
31 
32  // Check if the loop closures are interrobot.
33  // It is the case if {i,j} are elements of trajectory_robot1 and {k,l} are elements of trajectory_robot2.
34  // Or the inverse.
35  bool is_config_r12 = graph_utils::isInTrajectory(trajectory_robot1_, i) && graph_utils::isInTrajectory(trajectory_robot1_, j) &&
36  graph_utils::isInTrajectory(trajectory_robot2_, k) && graph_utils::isInTrajectory(trajectory_robot2_, l);
37  bool is_config_r21 = graph_utils::isInTrajectory(trajectory_robot2_, i) && graph_utils::isInTrajectory(trajectory_robot2_, j) &&
38  graph_utils::isInTrajectory(trajectory_robot1_, k) && graph_utils::isInTrajectory(trajectory_robot1_, l);
39  // Compute only if they are interrobot loop closures
40  if (is_config_r12 || is_config_r21) {
41  // Extract transforms
42  geometry_msgs::PoseWithCovariance abZik = (*transforms_interrobot_.transforms.find(loop_closure_1)).second.pose;
43  geometry_msgs::PoseWithCovariance abZjl = (*transforms_interrobot_.transforms.find(loop_closure_2)).second.pose;
44  geometry_msgs::PoseWithCovariance aXij, bXlk;
45  if (is_config_r12) {
46  aXij = composeOnTrajectory(i, j, 1);
47  bXlk = composeOnTrajectory(l, k, 2);
48  } else {
49  aXij = composeOnTrajectory(i, j, 2);
50  bXlk = composeOnTrajectory(l, k, 1);
51  }
52  // Compute the consistency pose (should be near Identity if consistent)
53  geometry_msgs::PoseWithCovariance consistency_pose = computeConsistencyPose(aXij, bXlk, abZik, abZjl);
54  // Compute the Mahalanobis distance
55  double distance = computeSquaredMahalanobisDistance(consistency_pose);
56  // Apply threshold on the chi-squared distribution
57  if (distance < threshold) {
58  consistency_matrix(u,v) = 1;
59  } else {
60  consistency_matrix(u,v) = 0;
61  }
62  }
63  }
64  v++;
65  }
66  u++;
67  }
68  return consistency_matrix;
69 }

84 double PairwiseConsistency::computeSquaredMahalanobisDistance(const geometry_msgs::PoseWithCovariance& transform) {
85  // Extraction of the covariance matrix
86  Eigen::Matrix<double, 6, 6> covariance_matrix;
87  for (int i = 0; i < 6; i++) {
88  for (int j = 0; j < 6; j++) {
89  covariance_matrix(i,j) = transform.covariance[i*6+j];
90  }
91  }
92 
93  // Extraction of the pose vector
94  Eigen::Matrix<double, 6, 1> pose_vector(6);
95  pose_vector(0) = transform.pose.position.x;
96  pose_vector(1) = transform.pose.position.y;
97  pose_vector(2) = transform.pose.position.z;
98  pose_vector(3) = transform.pose.orientation.x;
99  pose_vector(4) = transform.pose.orientation.y;
100  pose_vector(5) = transform.pose.orientation.z;
101 
102  // Computation of the squared Mahalanobis distance
103  double distance = pose_vector.transpose() * covariance_matrix * pose_vector;
104  return distance;
105 }
