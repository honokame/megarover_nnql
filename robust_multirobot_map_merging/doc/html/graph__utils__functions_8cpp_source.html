200 Trajectory buildTrajectory(const Transforms& transforms) {
201  // Initialization
202  Trajectory trajectory;
203  trajectory.start_id = transforms.start_id;
204  trajectory.end_id = transforms.end_id;
205  size_t current_pose_id = trajectory.start_id;
206  geometry_msgs::PoseWithCovariance temp_pose, total_pose;
207 
208  // Add first pose at the origin
209  graph_utils::TrajectoryPose current_pose;
210  current_pose.id = current_pose_id;
211  current_pose.pose.pose.orientation.w = 1;
212  temp_pose = current_pose.pose;
213  trajectory.trajectory_poses.insert(std::make_pair(current_pose_id, current_pose));
214 
215  // Initialization
216  std::pair<size_t, size_t> temp_pair = std::make_pair(current_pose_id, current_pose_id + 1);
217  auto temp_it = transforms.transforms.find(temp_pair);
218 
219  // Compositions in chain on the trajectory transforms.
220  while (temp_it != transforms.transforms.end() && !(*temp_it).second.is_loop_closure) {
221  graph_utils::poseCompose(temp_pose, (*temp_it).second.pose, total_pose);
222  temp_pose = total_pose;
223  current_pose_id++;
224  current_pose.id = current_pose_id;
225  current_pose.pose = total_pose;
226  trajectory.trajectory_poses.insert(std::make_pair(current_pose_id, current_pose));
227  temp_pair = std::make_pair(current_pose_id, current_pose_id + 1);
228  temp_it = transforms.transforms.find(temp_pair);
229  }
230 
231  return trajectory;
232 }
